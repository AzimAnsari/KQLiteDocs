Delete RETURNING

// DELETE FROM artists WHERE ArtistId > ?
// ORDER BY ArtistId DESC LIMIT ?
// RETURNING ArtistId;
val cursor = Artists
    .delete()
    .where {  it: Artists 
        it.artistId GT 1
    }
    .orderBy(Artists.artistId, Sort.DESC)
    .limit(3)
    .executeReturning(Artists.artistId)

val deleted = cursor
    .asSequence()
    .map { it[Artists.artistId] }
    .toList()

println("Artists deleted: ${deleted.joinToString()}")
cursor.close()
